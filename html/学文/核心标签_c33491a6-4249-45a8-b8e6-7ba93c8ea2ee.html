out标签

//defult    默认输出,escapeXml:   是否忽略XML特殊字符
<c:out value="<string>" default="<string>" escapeXml="<true|false>"/>

set标签

//target	要修改的属性所属的对象
//property	要修改的属性
<c:set var="salary" scope="session" value="${2000*2}"></c:set>

remove标签

<c:remove var="<string>" scope="<string>"/>

catch标签和if标签

<c:catch var ="catchException">
   <% int x = 5/0;%>
</c:catch>

<c:if test = "${catchException != null}">
   <p>异常为 : ${catchException} <br />
   发生了异常: ${catchException.message}</p>
</c:if>

choose,when,otherwise标签

<c:choose>
    <c:when test="<boolean>">
        ...
    </c:when>
    <c:when test="<boolean>">
        ...
    </c:when>
    ...
    ...
    <c:otherwise>
        ...
    </c:otherwise>
</c:choose>

import标签

//charEncoding	所引入的数据的字符编码集
//scope	var属性的作用域
//varReader	可选的用于提供java.io.Reader对象的变量
<c:import var="data" url="http://www.runoob.com"></c:import>

forEach标签

//item 	要被循环的信息
//step	每一次迭代的步长
//varStatus	代表循环状态的变量名称
<c:forEach var="i" begin="1" end="5">
   Item <c:out value="${i}"></c:out><p>
</c:forEach>

forTokens标签

<c:forTokens items="google,runoob,taobao" delims="," var="name">
   <c:out value="${name}"></c:out><p>
</c:forTokens>

url标签和param标签

<c:url var="myURL" value="main.jsp">
		<c:param name="name" value="Runoob"></c:param>
		<c:param name="url" value="www.runoob.com"></c:param>
	</c:url>
      <a href="/<c:out value="${myURL}"></a>">
 使用 &lt;c:param&gt; 为指定URL发送两个参数。</a>

redirect标签

<c:redirect url="http://www.runoob.com"></c:redirect>
